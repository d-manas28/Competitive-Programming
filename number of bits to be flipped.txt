#include<bits/stdc++.h>
using namespace std;
int countSetBits(int n)
{
    int setbits = 0;
    while(n>0){
        setbits = setbits + (n & 1);
        n = n>>1; 
    }
    return setbits;
}
int main()
{
    int n1,n2;
    cin>>n1>>n2;
    int n3 = n1 ^ n2;
    int setbits = countSetBits(n3);
    cout<<setbits<<endl;
    return 0;
}
